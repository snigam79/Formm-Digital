class WearableRepository( private val healthConnectClient: HealthConnectClient, private val api: FormmmApiService ) { // Request permissions — call from WearableHubFragment val PERMISSIONS = setOf( HealthPermission.getReadPermission(HeartRateVariabilityRmssdRecord::class), HealthPermission.getReadPermission(RestingHeartRateRecord::class), HealthPermission.getReadPermission(SleepSessionRecord::class), HealthPermission.getReadPermission(StepsRecord::class) ) // Read today's HRV — called in morning sync job suspend fun readTodayHRV(): Double? { val request = ReadRecordsRequest( recordType = HeartRateVariabilityRmssdRecord::class, timeRangeFilter = TimeRangeFilter.between( LocalDate.now().atStartOfDay().toInstant(ZoneOffset.UTC), Instant.now() ) ) val records = healthConnectClient.readRecords(request).records return records.lastOrNull()?.heartRateVariabilityMillis } // Read last night's sleep — returns SleepSummary suspend fun readLastNightSleep(): SleepSummary? { val yesterday = LocalDate.now().minusDays(1) val records = healthConnectClient.readRecords( ReadRecordsRequest( recordType = SleepSessionRecord::class, timeRangeFilter = TimeRangeFilter.between( yesterday.atTime(18, 0).toInstant(ZoneOffset.of("+05:30")), LocalDate.now().atTime(12, 0).toInstant(ZoneOffset.of("+05:30")) ) ) ).records return records.lastOrNull()?.let { parseSleepStages(it) } } // Sync to Formm backend — normalisation happens server-side suspend fun syncToBackend(hrv: Double?, sleep: SleepSummary?) { api.syncWearableData( WearableSyncRequest( deviceType = "health_connect", readings = buildList { hrv?.let { add(WearableReading("hrv_rmssd", it, Instant.now())) } sleep?.let { add(WearableReading("rem_pct", it.remPct, it.sessionEnd)) add(WearableReading("deep_min", it.deepMinutes, it.sessionEnd)) add(WearableReading("sleep_efficiency", it.efficiency, it.sessionEnd)) } } ) ) } }
⚠
IST timezone is critical for NCR users. Sleep sessions that start at 10pm IST and end at 6am IST must be queried with ZoneOffset.of("+05:30"). Using UTC will split overnight sleep sessions across two days and produce incorrect stage calculations.

class WearableSyncWorker(context: Context, params: WorkerParameters) : CoroutineWorker(context, params) { override suspend fun doWork(): Result { return try { val repo = /* inject WearableRepository */ val hrv = repo.readTodayHRV() val sleep = repo.readLastNightSleep() repo.syncToBackend(hrv, sleep) Result.success() } catch (e: Exception) { Result.retry() // retry up to 3x with backoff } } } // Schedule in Application.onCreate() fun scheduleWearableSync(context: Context) { val istNow = ZonedDateTime.now(ZoneId.of("Asia/Kolkata")) val next6am = istNow .withHour(6).withMinute(0).withSecond(0) .let { if (it.isBefore(istNow)) it.plusDays(1) else it } val delay = Duration.between(istNow, next6am) WorkManager.getInstance(context).enqueueUniquePeriodicWork( "wearable_sync", ExistingPeriodicWorkPolicy.UPDATE, PeriodicWorkRequestBuilder<WearableSyncWorker>(24, TimeUnit.HOURS) .setInitialDelay(delay) .setConstraints(Constraints(requiresNetworkConnectivity = true)) .build() ) }

@app.route("/api/terra/webhook", methods=["POST"]) def terra_webhook(): payload = request.get_json() user_id = get_formm_user_id(payload["user"]["user_id"]) device = payload["user"]["provider"].lower() # "whoop", "oura", "garmin" for data_type, data_list in payload.get("data", {}).items(): for record in data_list: process_wearable_record(user_id, device, data_type, record) return jsonify({"status": "ok"}), 200 def process_wearable_record(user_id, device, data_type, record): # Extract readings based on Terra's normalised schema readings = extract_readings(device, data_type, record) for metric, value, timestamp in readings: # Store raw reading store_raw_reading(user_id, device, metric, value, timestamp) # Compute z-score IMMEDIATELY — never store without normalising update_normalisation(user_id, device, metric) # Trigger inference check asynchronously queue_inference_check.delay(user_id) # Celery task def extract_readings(device, data_type, record): """Map Terra's normalised schema to Formm internal metrics.""" readings = [] if data_type == "daily": if hrv := record.get("hrv_data", {}).get("rmssd_sample_mean"): readings.append(("hrv_rmssd", hrv, record["date"])) if rhr := record.get("heart_rate_data", {}).get("resting_hr_bpm"): readings.append(("resting_hr", rhr, record["date"])) elif data_type == "sleep": stages = record.get("sleep_durations_data", {}) if deep := stages.get("asleep_stage_3_seconds"): readings.append(("deep_sleep_min", deep/60, record["end_time"])) if rem := stages.get("asleep_stage_rem_seconds"): total = record.get("sleep_durations_data", {}).get("total_sleep_seconds", 1) readings.append(("rem_pct", (rem/total)*100, record["end_time"])) elif data_type == "activity" and device == "whoop": if strain := record.get("strain_data", {}).get("strain_level"): readings.append(("whoop_strain", strain, record["start_time"])) return readings
05 — Android Code

Z-Score Normalisation

The most critical component in the entire system. Build this before any inference logic. Get it wrong and every downstream insight is wrong.

🚫
Never compare raw HRV values across devices. Oura measures HRV over the entire night in 5-min windows. WHOOP weights toward last slow-wave sleep. Garmin uses a 30-min window of unknown timing. Apple Watch uses SDNN (not RMSSD). A 55ms reading means something different from each device. Only z-scores (deviation from personal baseline) are device-comparable.
Pythonservices/normalisation_service.py
def update_normalisation(user_id: int, device: str, metric: str): """Recompute 7-day baseline and z-score after each new reading. Called immediately after every raw reading is stored.""" # Get last 7 readings for this user/device/metric readings = db.query(""" SELECT value FROM wearable_readings WHERE user_id = %s AND device = %s AND metric = %s AND timestamp >= NOW() - INTERVAL '7 days' ORDER BY timestamp DESC LIMIT 7 """, [user_id, device, metric]) if len(readings) < 7: # Insufficient data — mark as "building baseline" # DO NOT compute z-score from fewer than 7 readings db.update_baseline(user_id, device, metric, mean=None, sd=None, status="building") return values = [r["value"] for r in readings] mean = statistics.mean(values) sd = statistics.stdev(values) if sd == 0: # All readings identical — skip (would cause div by zero) return # Compute z-score for today's reading today_value = readings[0]["value"] z_score = (today_value - mean) / sd # Store baseline and today's z-score db.update_baseline(user_id, device, metric, mean=mean, sd=sd, status="active") db.update_today_z_score(user_id, device, metric, z_score=z_score, reading_count=len(readings)) def get_primary_hrv_z_score(user_id: int) -> Optional[float]: """Get today's HRV z-score. Uses Oura if available (highest accuracy), falls back to WHOOP, then Health Connect, then Formm camera rPPG.""" device_priority = ["oura", "whoop", "health_connect", "formm_rpgg"] for device in device_priority: z = db.get_today_z_score(user_id, device, "hrv_rmssd") if z is not None: return z, device return None, None
💡
Device priority for HRV: Oura Gen 3/4 → WHOOP 4.0 → Health Connect → Formm camera rPPG. This ordering follows the 2025 Dial et al. peer-reviewed accuracy rankings. Always return which device sourced the HRV to the frontend so it can display the source badge.

Inference Engine

All inferences are server-side. Rule-based for V1 — fast to ship, auditable, and explainable. ML layer added in V2 once data volumes justify it.

Pythonservices/inference_engine.py
class InferenceEngine: def run_daily_inferences(self, user_id: int) -> List[Insight]: """Run all V1 inferences for a user. Called at 6am daily per user timezone.""" insights = [] # Fetch all normalised data hrv_z, hrv_device = get_primary_hrv_z_score(user_id) sleep = get_last_night_sleep(user_id) acwr = get_current_acwr(user_id) form_trend = get_form_score_trend(user_id, days=7) hrv_trend = get_hrv_trend_7d(user_id) # ── INFERENCE 1: HRV-adjusted session target ────────────── if hrv_z is not None: insights.append(self._hrv_session_target(user_id, hrv_z, hrv_device)) # ── INFERENCE 2: Red alert convergence ──────────────────── if all([hrv_z, acwr, form_trend]): alert = self._check_convergence_alert(hrv_z, acwr, form_trend) if alert: insights.append(alert) # ── INFERENCE 3: Fatigue vs technique diagnosis ──────────── if hrv_trend and form_trend: diag = self._fatigue_vs_technique(hrv_trend, form_trend) if diag: insights.append(diag) # ── INFERENCE 4: Illness early warning ──────────────────── skin_temp_z = get_skin_temp_z(user_id) resp_z = get_resp_rate_z(user_id) if skin_temp_z and resp_z: illness = self._illness_flag(skin_temp_z, resp_z) if illness: insights.append(illness) return [i for i in insights if i is not None] def _check_convergence_alert(self, hrv_z, acwr, form_trend) -> Optional[Insight]: """Red alert: three signals converging = mandatory rest.""" form_slope = linear_regression_slope(form_trend[-3:]) if (hrv_z < -1.5 and acwr > 1.5 and form_slope < -0.10): return Insight( type = "convergence_risk_high", severity = "mandatory", headline = "Three risk signals converging", body = f"HRV is {abs(hrv_z):.1f} SD below baseline, ACWR is " f"{acwr:.2f}, and form quality dropped " f"{abs(form_slope)*100:.0f}% over 3 sessions. " "Mandatory rest today and tomorrow.", sources = ["hrv", "acwr", "form_score"], action = None # no action — rest is the action ) return None def _fatigue_vs_technique(self, hrv_trend, form_trend) -> Optional[Insight]: """The most clinically important inference: diagnose WHY form is dropping.""" form_slope = linear_regression_slope(form_trend[-5:]) hrv_slope = linear_regression_slope(hrv_trend[-7:]) if form_slope >= -0.03: return None # form not declining — inference irrelevant if hrv_slope < -0.5: return Insight( type="fatigue_driven_degradation", severity="amber", headline="Form dropping due to fatigue — not technique", body="Your form score has dropped alongside your HRV. " "This is fatigue-driven degradation. Rest first — " "more practice won't help until you recover.", sources=["hrv", "form_score"] ) elif hrv_slope > -0.2: return Insight( type="technique_regression", severity="info", headline="Form gap detected — recovery is fine", body="Your form score is dropping but HRV is stable. " "This is a genuine technique gap. Focused practice " "(not rest) is the right response.", sources=["hrv", "form_score"] ) return None

15 — Ship It

Edge Cases & Risks

The things that will bite you if you don't handle them before launch.

R1
OEM battery optimisation kills WorkManager (Xiaomi, OnePlus, Realme)
The most common failure mode for Indian Android deployments. These OEMs aggressively kill background processes. Morning sync job will silently fail. Add to onboarding: a guided "Allow background activity" flow that deep-links to the OEM's specific battery settings screen. Test explicitly on Xiaomi MIUI and OnePlus OxygenOS before launch.
R2
User has both WHOOP and Oura — which HRV to use?
Always use Oura as primary HRV source (CCC 0.99 vs WHOOP CCC 0.94 per 2025 peer-reviewed study). WHOOP as secondary/validation. If both exist and z-scores diverge by >1.5: flag to user. Never average them — they use different algorithms. Display both values in expanded readiness breakdown with "primary source" label.
R3
WHOOP strain for "yesterday" vs session-specific strain
WHOOP strain accumulates throughout the day — there's no per-exercise strain split. Match WHOOP strain to Formm sessions by date, not by timestamp. Edge case: user uploads form video 2 days after session — use video's session_date for strain lookup, not upload timestamp. If multiple sessions in one day, allocate strain proportionally by session load AU.
R4
NCR summer heat (+40°C) affects skin temp baselines
Oura and WHOOP skin temperature sensors are affected by ambient temperature. In Indian summers, baseline skin temp will be elevated compared to winter. Use a 14-day rolling baseline for skin temp (not 7-day) to reduce seasonal drift. Add a note in the inference output when skin temp z-score is borderline: "Warm ambient temperatures can affect this reading."
R5
User connects wearable mid-inference cycle
User connects Oura on Day 8. They have 8 days of Oura data but zero previously. The normalisation pipeline needs 7 days minimum — on Day 8, they have exactly 7. But: the first 7-day baseline has high variance (user's body is still adapting to tracking). Suppress inferences that require z-score precision for the first 14 days after connection, not 7. Show "calibrating" state in UI for this window.
R6
Terra webhook delivery failures / duplicate events
Terra may retry webhooks on timeout. Implement idempotency keys on /api/terra/webhook: store (terra_user_id, data_type, timestamp) as unique key. Reject duplicates with 200 OK (not 4xx — Terra will retry on non-200). Handle Terra API rate limits: queue large historical backfills (first-connect 30-day backfill) as async batch jobs, not synchronous webhook processing.
R7
DPDP Act 2023 — health data is sensitive personal data
India's Digital Personal Data Protection Act classifies biometric data as sensitive. Explicit, specific consent required before collection. Store consent record with timestamp, version, and scope. Re-consent required if scope changes. Data localisation: health data must be stored in India or in a jurisdiction with equivalent protection. Add "Delete all wearable data" with immediate effect — not a 30-day delay.
⚡
The most important thing to remember: The product must be fully valuable without any wearable connected. The Formm camera rPPG already provides the core HRV readiness signal with no hardware required. Wearables enhance accuracy and unlock additional inferences — they never gate core product value. Design every empty state and fallback to communicate capability, not incompleteness.
